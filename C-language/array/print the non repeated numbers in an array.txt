//method 1
#include<stdio.h>
main()
{
	int i,ele,j,k,c,a[]={0,3,1,0,5,1,2,0,4,5};
	ele=sizeof a/sizeof a[0];
	for(i=0;i<ele;i++)
	{c=0;
		for(j=i+1;j<ele;j++)
		{
			if(a[i]==a[j])
			{
				c++;
				for(k=j;k<ele;k++)
					a[k]=a[k+1];
				ele--;
			}
		}
		if(c==0)
			printf("%d ",a[i]);

	}
}
