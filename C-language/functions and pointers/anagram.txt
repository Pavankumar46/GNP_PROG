void upper(char *p)
{
	int i;
	for(i=0;p[i];i++)
		if(p[i]>='a'&&p[i]<='z')
			p[i]-=32;
}
int anag(char *p,char *q)
{
	int c,i,j,r;
	for(i=0;p[i];i++)
	{
		if(p[i]>='A'&&p[i]<='Z')
		{
			c=0;
			r=p[i];		
			for(j=0;p[j];j++)
				if(p[j]==r)
					c++;
			for(j=0;q[j];j++)
				if(q[j]==r)
					c--;
			if(c!=0)
			{
			return 0;
			break;
			}
		}
	}
