void arrreverse(char *p,int j,int i)
{
	int k,t;char ch;
	if(i<j)
	{
		ch=*(p+i);
		*(p+i)=*(p+j);
		*(p+j)=ch;
		arrreverse(p,j-1,i+1);
	}
}
